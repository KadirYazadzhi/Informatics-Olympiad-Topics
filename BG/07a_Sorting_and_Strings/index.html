    void bubbleSort(vector<int>& arr) {
    int n = arr.size();
    for (int i = 0; i < n - 1; i++) {
        bool swapped = false;
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                swap(arr[j], arr[j + 1]);
                swapped = true;
            }
        }
        if (!swapped) break; // –û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è
    }
}

    2.1. –°–æ—Ä—Ç–∏—Ä–∞–Ω–µ —á—Ä–µ–∑ —Å–ª–∏–≤–∞–Ω–µ (Merge Sort)

    –ö–ª–∞—Å–∏—á–µ—Å–∫–∏ –ø—Ä–∏–º–µ—Ä –∑–∞ —Å—Ç—Ä–∞—Ç–µ–≥–∏—è—Ç–∞ "–†–∞–∑–¥–µ–ª—è–π –∏ –≤–ª–∞–¥–µ–π" (Divide and Conquer). 1. –†–∞–∑–¥–µ–ª—è–Ω–µ: –ú–∞—Å–∏–≤—ä—Ç —Å–µ –¥–µ–ª–∏ –Ω–∞ –¥–≤–µ –ø–æ–ª–æ–≤–∏–Ω–∏. 2. –í–ª–∞–¥–µ–µ–Ω–µ: –í—Å—è–∫–∞ –ø–æ–ª–æ–≤–∏–Ω–∞ —Å–µ —Å–æ—Ä—Ç–∏—Ä–∞ —Ä–µ–∫—É—Ä—Å–∏–≤–Ω–æ. 3. –°–ª–∏–≤–∞–Ω–µ: –î–≤–µ—Ç–µ —Å–æ—Ä—Ç–∏—Ä–∞–Ω–∏ –ø–æ–ª–æ–≤–∏–Ω–∏ —Å–µ –æ–±–µ–¥–∏–Ω—è–≤–∞—Ç (merge) –≤ –µ–¥–∏–Ω —Å–æ—Ä—Ç–∏—Ä–∞–Ω –º–∞—Å–∏–≤.

    • –°—Ç–∞–±–∏–ª–Ω–æ—Å—Ç: –î–∞ (–∑–∞–ø–∞–∑–≤–∞ —Ä–µ–¥–∞ –Ω–∞ —Ä–∞–≤–Ω–∏—Ç–µ –µ–ª–µ–º–µ–Ω—Ç–∏).
    • –ü–∞–º–µ—Ç: –ò–∑–∏—Å–∫–≤–∞ \(O(N)\) –¥–æ–ø—ä–ª–Ω–∏—Ç–µ–ª–Ω–∞ –ø–∞–º–µ—Ç.
    void merge(vector<int>& arr, int l, int m, int r) {
    vector<int> temp;
    int i = l, j = m + 1;
    while (i <= m && j <= r) {
        if (arr[i] <= arr[j]) temp.push_back(arr[i++]);
        else temp.push_back(arr[j++]);
    }
    while (i <= m) temp.push_back(arr[i++]);
    while (j <= r) temp.push_back(arr[j++]);
    for (int k = 0; k < temp.size(); k++) arr[l + k] = temp[k];
}

void mergeSort(vector<int>& arr, int l, int r) {
    if (l < r) {
        int m = l + (r - l) / 2;
        mergeSort(arr, l, m);
        mergeSort(arr, m + 1, r);
        merge(arr, l, m, r);
    }
}

    –°—ä—â–æ –∏–∑–ø–æ–ª–∑–≤–∞ "–†–∞–∑–¥–µ–ª—è–π –∏ –≤–ª–∞–¥–µ–π", –Ω–æ —Ä–∞–∑–¥–µ–ª—è–Ω–µ—Ç–æ —Å—Ç–∞–≤–∞ —á—Ä–µ–∑ Pivot (–æ–ø–æ—Ä–µ–Ω –µ–ª–µ–º–µ–Ω—Ç). –í—Å–∏—á–∫–∏ –ø–æ-–º–∞–ª–∫–∏ –æ—Ç –ø–∏–≤–æ—Ç–∞ –æ—Ç–∏–≤–∞—Ç –≤–ª—è–≤–æ, –∞ –ø–æ-–≥–æ–ª–µ–º–∏—Ç–µ ‚Äì –≤–¥—è—Å–Ω–æ. * –ë—ä—Ä–∑–∏–Ω–∞: –í –ø—Ä–∞–∫—Ç–∏–∫–∞—Ç–∞ –µ –ø–æ-–±—ä—Ä–∑ –æ—Ç Merge Sort, –∑–∞—â–æ—Ç–æ —Ä–∞–±–æ—Ç–∏ "in-place" (–±–µ–∑ –¥–æ–ø—ä–ª–Ω–∏—Ç–µ–ª–Ω–∞ –ø–∞–º–µ—Ç) –∏ –∏–º–∞ –¥–æ–±—Ä–∞ –∫–µ—à-–ª–æ–∫–∞–ª–Ω–æ—Å—Ç. * –ù–∞–π-–ª–æ—à —Å–ª—É—á–∞–π: \(O(N^2)\), –∞–∫–æ –ø–∏–≤–æ—Ç—ä—Ç –µ –ª–æ—à–æ –∏–∑–±—Ä–∞–Ω (–Ω–∞–ø—Ä. —Å–æ—Ä—Ç–∏—Ä–∞–Ω –º–∞—Å–∏–≤). * std::sort –≤ C++ –∏–∑–ø–æ–ª–∑–≤–∞ —Ö–∏–±—Ä–∏–¥–µ–Ω –≤–∞—Ä–∏–∞–Ω—Ç (Introsort: QuickSort + HeapSort), –∑–∞ –¥–∞ –≥–∞—Ä–∞–Ω—Ç–∏—Ä–∞ \(O(N \log N)\).

    3.1. Counting Sort (–°–æ—Ä—Ç–∏—Ä–∞–Ω–µ —á—Ä–µ–∑ –±—Ä–æ–µ–Ω–µ)
