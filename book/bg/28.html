string getPermutation(int n, int k) {
    vector<int> numbers;
    long long fact = 1;
    for (int i = 1; i <= n; i++) {
        numbers.push_back(i);
        if (i < n) fact *= i;
    }
    k--; // 0-based index
    string ans = "";
    for (int i = n - 1; i >= 0; i--) {
        int idx = k / fact;
        ans += to_string(numbers[idx]);
        numbers.erase(numbers.begin() + idx);
        if (i > 0) {
            k %= fact;
            fact /= i;
        }
    }
    return ans;
}
