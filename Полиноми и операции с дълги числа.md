struct BigInt {
    std::vector<int> digits;
    static const int BASE = 10;

    BigInt(std::string s) {
        for (int i = s.size() - 1; i >= 0; i--) {
            digits.push_back(s[i] - '0');
        }
    }

    void add(const BigInt& other) {
        int carry = 0;
        for (size_t i = 0; i < std::max(digits.size(), other.digits.size()) || carry; i++) {
            if (i == digits.size()) digits.push_back(0);
            int cur = digits[i] + carry + (i < other.digits.size() ? other.digits[i] : 0);
            digits[i] = cur % BASE;
            carry = cur / BASE;
        }
    }

    void print() {
        if (digits.empty()) { std::cout << 0; return; }
        for (int i = digits.size() - 1; i >= 0; i--) std::cout << digits[i];
        std::cout << std::endl;
    }
};
