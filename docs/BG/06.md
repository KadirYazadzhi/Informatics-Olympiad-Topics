    int count = 0, candidate = -1;
    
    // Boyer-Moore Voting Algorithm
    for (int i = 0; i < n; i++) {
        if (count == 0) {
            candidate = arr[i];
            count = 1;
        } else if (arr[i] == candidate) {
            count++;
        } else {
            count--;
        }
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ –∫–∞–Ω–¥–∏–¥–∞—Ç—ä—Ç –Ω–∞–∏—Å—Ç–∏–Ω–∞ –µ –º–Ω–æ–∑–∏–Ω—Å—Ç–≤–æ

        if (arr[i] == candidate) count++;
    }
