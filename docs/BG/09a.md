void add(BigInt &a, const BigInt &b) {
    int carry = 0;
    for (size_t i = 0; i < max(a.d.size(), b.d.size()) || carry; ++i) {
        if (i == a.d.size()) a.d.push_back(0);
        long long cur = a.d[i] + carry + (i < b.d.size() ? b.d[i] : 0);
