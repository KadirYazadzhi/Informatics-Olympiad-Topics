    for (int i = 0; i < nums.size(); i++) {
        // –ü—Ä–µ–º–∞—Ö–≤–∞ –µ–ª–µ–º–µ–Ω—Ç–∏ –∏–∑–≤—ä–Ω –ø—Ä–æ–∑–æ—Ä–µ—Ü–∞
        if (!dq.empty() && dq.front() <= i - k) {
            dq.pop_front();
        }
        
        // –ü—Ä–µ–º–∞—Ö–≤–∞ –ø–æ-–º–∞–ª–∫–∏ –µ–ª–µ–º–µ–Ω—Ç–∏ –æ—Ç –∫—Ä–∞—è
        while (!dq.empty() && nums[dq.back()] < nums[i]) {
            dq.pop_back();
        }
        
        dq.push_back(i);
        
        // –î–æ–±–∞–≤—è —Ä–µ–∑—É–ª—Ç–∞—Ç —Å–ª–µ–¥ –ø—ä—Ä–≤–∏—Ç–µ k-1 –µ–ª–µ–º–µ–Ω—Ç–∞
        if (i >= k - 1) {
            result.push_back(nums[dq.front()]);
        }
    }
    
    return result;
}

int main() {
