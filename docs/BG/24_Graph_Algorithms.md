### 2.1. –ê–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –î–µ–π–∫—Å—Ç—Ä–∞ (Dijkstra)
–ê–ª–≥–æ—Ä–∏—Ç—ä–º—ä—Ç –Ω–∞ –î–µ–π–∫—Å—Ç—Ä–∞ –Ω–∞–º–∏—Ä–∞ –Ω–∞–π-–∫—Ä–∞—Ç–∫–∏—è –ø—ä—Ç –æ—Ç –Ω–∞—á–∞–ª–µ–Ω –≤—Ä—ä—Ö –¥–æ –≤—Å–∏—á–∫–∏ –æ—Å—Ç–∞–Ω–∞–ª–∏ –≤—ä—Ä—Ö–æ–≤–µ –≤ –≥—Ä–∞—Ñ —Å **–Ω–µ–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏** —Ç–µ–≥–ª–∞. –ò–∑–ø–æ–ª–∑–≤–∞ –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç–Ω–∞ –æ–ø–∞—à–∫–∞ –∑–∞ –≤–∏–Ω–∞–≥–∏ –¥–∞ –æ–±—Ä–∞–±–æ—Ç–≤–∞ –≤—ä—Ä—Ö–∞ —Å –Ω–∞–π-–º–∞–ª–∫–æ —Ä–∞–∑—Å—Ç–æ—è–Ω–∏–µ.

**–ò–¥–µ—è:** –ü–æ—Å—Ç–µ–ø–µ–Ω–Ω–æ "—Ä–µ–ª–∞–∫—Å–∏—Ä–∞–º–µ" —Ä–∞–∑—Å—Ç–æ—è–Ω–∏—è—Ç–∞, –≤–∏–Ω–∞–≥–∏ –∏–∑–±–∏—Ä–∞–π–∫–∏ –Ω–∞–π-–±–ª–∏–∑–∫–∏—è –Ω–µ–æ–±—Ä–∞–±–æ—Ç–µ–Ω –≤—Ä—ä—Ö.

**–ü—ä–ª–Ω–∞ –∏–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è —Å –≤—ä–∑—Å—Ç–∞–Ω–æ–≤—è–≤–∞–Ω–µ –Ω–∞ –ø—ä—Ç:**
```cpp

vector<pair<int, int>> adj[MAXN];  // {—Å—ä—Å–µ–¥, —Ç–µ–≥–ª–æ}

void dijkstra(int start) {

    fill(parent, parent + n, -1);
    dist[start] = 0;
    
    // Min-heap: {—Ä–∞–∑—Å—Ç–æ—è–Ω–∏–µ, –≤—Ä—ä—Ö}

                   vector<pair<long long, int>>,

        auto [d, u] = pq.top(); 
        pq.pop();
        
        // –ü—Ä–æ–ø—É—Å–∫–∞–º–µ –æ—Å—Ç–∞—Ä–µ–ª–∏ –∑–∞–ø–∏—Å–∏

                dist[v] = dist[u] + w;
                parent[v] = u;
                pq.push({dist[v], v});
            }
        }
    }
}

    if (dist[end] == INF) return {};  // –ù—è–º–∞ –ø—ä—Ç
    
    vector<int> path;
    for (int v = end; v != -1; v = parent[v]) {
        path.push_back(v);
    }
    reverse(path.begin(), path.end());
    return path;
}
```

**–í–∞–∂–Ω–æ:** –î–µ–π–∫—Å—Ç—Ä–∞ **–Ω–µ —Ä–∞–±–æ—Ç–∏** –∫–æ—Ä–µ–∫—Ç–Ω–æ –ø—Ä–∏ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏ —Ç–µ–≥–ª–∞!

### 2.2. –ê–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –ë–µ–ª–º–∞–Ω-–§–æ—Ä–¥ (Bellman-Ford)
Bellman-Ford —Ä–∞–±–æ—Ç–∏ —Å **–æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏** —Ç–µ–≥–ª–∞ –∏ –º–æ–∂–µ –¥–∞ –æ—Ç–∫—Ä–∏–≤–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏ —Ü–∏–∫–ª–∏. –ê–ª–≥–æ—Ä–∏—Ç—ä–º—ä—Ç —Ä–µ–ª–∞–∫—Å–∏—Ä–∞ –≤—Å–∏—á–∫–∏ —Ä–µ–±—Ä–∞ $V-1$ –ø—ä—Ç–∏.

**–ò–¥–µ—è:** –ê–∫–æ –∏–º–∞ –ø—ä—Ç —Å –Ω–∞–π-–º–Ω–æ–≥–æ $k$ —Ä–µ–±—Ä–∞, —Ç–æ —Å–ª–µ–¥ $k$ –∏—Ç–µ—Ä–∞—Ü–∏–∏ —Ä–∞–∑—Å—Ç–æ—è–Ω–∏–µ—Ç–æ —â–µ –±—ä–¥–µ –∫–æ—Ä–µ–∫—Ç–Ω–æ.

**–ü—ä–ª–Ω–∞ –∏–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è:**
```cpp
struct Edge {
    int from, to;
    long long weight;
};

vector<Edge> edges;
long long dist[MAXN];
int parent[MAXN];

bool bellmanFord(int start, int n) {
    fill(dist, dist + n, INF);
    fill(parent, parent + n, -1);
    dist[start] = 0;
    
    // –†–µ–ª–∞–∫—Å–∞—Ü–∏—è V-1 –ø—ä—Ç–∏
    for (int i = 0; i < n - 1; i++) {
        for (auto& e : edges) {
            if (dist[e.from] != INF && dist[e.from] + e.weight < dist[e.to]) {
                dist[e.to] = dist[e.from] + e.weight;
                parent[e.to] = e.from;
            }
        }
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –∑–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω —Ü–∏–∫—ä–ª
    for (auto& e : edges) {
        if (dist[e.from] != INF && dist[e.from] + e.weight < dist[e.to]) {
            return false;  // –ò–º–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω —Ü–∏–∫—ä–ª
        }
    }
    
    return true;  // –ù—è–º–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω —Ü–∏–∫—ä–ª
}

void findNegativeCycleNodes(int n) {
    bool relaxed[MAXN] = {};
    
    // –û—â–µ V –∏—Ç–µ—Ä–∞—Ü–∏–∏ - —Ä–∞–∑—Å—Ç–æ—è–Ω–∏—è—Ç–∞ –ø—Ä–æ–¥—ä–ª–∂–∞–≤–∞—Ç –¥–∞ –Ω–∞–º–∞–ª—è–≤–∞—Ç
    // —Å–∞–º–æ –∑–∞ –≤—ä—Ä—Ö–æ–≤–µ –≤/–¥–æ—Å—Ç–∏–∂–∏–º–∏ –æ—Ç –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω —Ü–∏–∫—ä–ª
    for (int i = 0; i < n; i++) {
        for (auto& e : edges) {
            if (dist[e.from] != INF && dist[e.from] + e.weight < dist[e.to]) {
                dist[e.to] = -INF;  // –ú–∞—Ä–∫–∏—Ä–∞–º–µ –∫–∞—Ç–æ "–±–µ–∑–∫—Ä–∞–π–Ω–æ –º–∞–ª–∫–æ"
                relaxed[e.to] = true;
            }
        }
    }
}
```

**–°–ª–æ–∂–Ω–æ—Å—Ç:** $O(V \cdot E)$ - –ø–æ-–±–∞–≤–µ–Ω –æ—Ç –î–µ–π–∫—Å—Ç—Ä–∞, –Ω–æ –ø–æ-—É–Ω–∏–≤–µ—Ä—Å–∞–ª–µ–Ω.

**–û–ø—Ç–∏–º–∏–∑–∞—Ü–∏—è:** –ú–æ–∂–µ–º –¥–∞ —Å–ø—Ä–µ–º –ø–æ-—Ä–∞–Ω–æ, –∞–∫–æ –≤ –Ω—è–∫–æ—è –∏—Ç–µ—Ä–∞—Ü–∏—è –Ω—è–º–∞ —Ä–µ–ª–∞–∫—Å–∞—Ü–∏—è.

**–ü—Ä–∏–ª–æ–∂–µ–Ω–∏–µ:** –ó–∞–¥–∞—á–∏ —Å –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏ —Ç–µ–≥–ª–∞, –∞—Ä–±–∏—Ç—Ä–∞–∂ –≤—ä–≤ —Ñ–∏–Ω–∞–Ω—Å–∏.

### 2.3. –ê–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –§–ª–æ–π–¥-–£–æ—Ä—à–∞–ª (Floyd-Warshall)
Floyd-Warshall –Ω–∞–º–∏—Ä–∞ **–Ω–∞–π-–∫—Ä–∞—Ç–∫–∏—è –ø—ä—Ç –º–µ–∂–¥—É –≤—Å–∏—á–∫–∏ –¥–≤–æ–π–∫–∏** –≤—ä—Ä—Ö–æ–≤–µ —á—Ä–µ–∑ –¥–∏–Ω–∞–º–∏—á–Ω–æ –ø—Ä–æ–≥—Ä–∞–º–∏—Ä–∞–Ω–µ. –ò–∑–ø–æ–ª–∑–≤–∞ —Å–µ –∑–∞ –º–∞–ª–∫–∏ –≥—Ä–∞—Ñ–∏ (–¥–æ $\approx$ 400-500 –≤—ä—Ä—Ö–∞).

**–ò–¥–µ—è:** $dp[k][i][j]$ = –Ω–∞–π-–∫—Ä–∞—Ç—ä–∫ –ø—ä—Ç –æ—Ç $i$ –¥–æ $j$, –∏–∑–ø–æ–ª–∑–≤–∞–π–∫–∏ —Å–∞–º–æ –≤—ä—Ä—Ö–æ–≤–µ $0...k$ –∫–∞—Ç–æ –º–µ–∂–¥–∏–Ω–Ω–∏. –ú–æ–∂–µ –¥–∞ —Å–µ –æ–ø—Ç–∏–º–∏–∑–∏—Ä–∞ –¥–æ 2D –º–∞—Å–∏–≤.

**–ü—ä–ª–Ω–∞ –∏–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è:**
```cpp
const long long INF = 1e18;
long long dist[MAXN][MAXN];
int next[MAXN][MAXN];  // –ó–∞ –≤—ä–∑—Å—Ç–∞–Ω–æ–≤—è–≤–∞–Ω–µ –Ω–∞ –ø—ä—Ç

void floydWarshall(int n) {
    // –ò–Ω–∏—Ü–∏–∞–ª–∏–∑–∞—Ü–∏—è
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            if (i == j) {
                dist[i][j] = 0;
            } else {
                dist[i][j] = INF;
            }
            next[i][j] = -1;
        }
    }
    
    // –î–æ–±–∞–≤—è–º–µ —Ä–µ–±—Ä–∞—Ç–∞
    for (auto [u, v, w] : edges) {
        dist[u][v] = w;
        next[u][v] = v;
    }
    
    // –û—Å–Ω–æ–≤–µ–Ω –∞–ª–≥–æ—Ä–∏—Ç—ä–º
    for (int k = 0; k < n; k++) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                if (dist[i][k] != INF && dist[k][j] != INF) {
                    if (dist[i][k] + dist[k][j] < dist[i][j]) {
                        dist[i][j] = dist[i][k] + dist[k][j];
                        next[i][j] = next[i][k];
                    }
                }
            }
        }
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –∑–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏ —Ü–∏–∫–ª–∏
    for (int i = 0; i < n; i++) {
        if (dist[i][i] < 0) {
            // –ò–º–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–µ–Ω —Ü–∏–∫—ä–ª, –¥–æ—Å—Ç–∏–∂–∏–º –æ—Ç i
        }
    }
}

// –í—ä–∑—Å—Ç–∞–Ω–æ–≤—è–≤–∞–Ω–µ –Ω–∞ –ø—ä—Ç –æ—Ç u –¥–æ v
vector<int> reconstructPath(int u, int v) {
    if (next[u][v] == -1) return {};  // –ù—è–º–∞ –ø—ä—Ç
    
    vector<int> path = {u};
    while (u != v) {
        u = next[u][v];
        path.push_back(u);
    }
    return path;
}
```

**–°–ª–æ–∂–Ω–æ—Å—Ç:** $O(V^3)$ - —Ç—Ä–∏ –≤–ª–æ–∂–µ–Ω–∏ —Ü–∏–∫—ä–ª–∞.

**–ü—Ä–∏–ª–æ–∂–µ–Ω–∏—è:**
- –ù–∞–º–∏—Ä–∞–Ω–µ –Ω–∞ —Ç—Ä–∞–Ω–∑–∏—Ç–∏–≤–Ω–æ –∑–∞—Ç–≤–∞—Ä—è–Ω–µ –Ω–∞ –≥—Ä–∞—Ñ
- –û—Ç–∫—Ä–∏–≤–∞–Ω–µ –Ω–∞ –æ—Ç—Ä–∏—Ü–∞—Ç–µ–ª–Ω–∏ —Ü–∏–∫–ª–∏
- –ú–∏–Ω–∏–º–∞–∫—Å/–º–∞–∫—Å–∏–º–∏–Ω –ø—ä—Ç–∏—â–∞ (—Å –º–æ–¥–∏—Ñ–∏–∫–∞—Ü–∏—è)

## 3. –ú–∏–Ω–∏–º–∞–ª–Ω–æ –ü–æ–∫—Ä–∏–≤–∞—â–æ –î—ä—Ä–≤–æ (MST)

–ú–∏–Ω–∏–º–∞–ª–Ω–æ—Ç–æ –ø–æ–∫—Ä–∏–≤–∞—â–æ –¥—ä—Ä–≤–æ –µ –ø–æ–¥–º–Ω–æ–∂–µ—Å—Ç–≤–æ –æ—Ç —Ä–µ–±—Ä–∞—Ç–∞, –∫–æ–µ—Ç–æ —Å–≤—ä—Ä–∑–≤–∞ –≤—Å–∏—á–∫–∏ –≤—ä—Ä—Ö–æ–≤–µ —Å –º–∏–Ω–∏–º–∞–ª–Ω–∞ –æ–±—â–∞ —Å—É–º–∞ –Ω–∞ —Ç–µ–≥–ª–∞—Ç–∞.

### 3.1. –ê–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –ö—Ä—É—Å–∫–∞–ª (Kruskal)
–ö—Ä—É—Å–∫–∞–ª —Å–æ—Ä—Ç–∏—Ä–∞ –≤—Å–∏—á–∫–∏ —Ä–µ–±—Ä–∞ –ø–æ —Ç–µ–≥–ª–æ –∏ –≥–∏ –¥–æ–±–∞–≤—è –µ–¥–Ω–æ –ø–æ –µ–¥–Ω–æ, –∞–∫–æ –Ω–µ –æ–±—Ä–∞–∑—É–≤–∞—Ç —Ü–∏–∫—ä–ª. –ò–∑–ø–æ–ª–∑–≤–∞ Disjoint Set Union (DSU) –∑–∞ –µ—Ñ–µ–∫—Ç–∏–≤–Ω–∞ –ø—Ä–æ–≤–µ—Ä–∫–∞.

**–ü—ä–ª–Ω–∞ –∏–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è:**
```cpp
struct Edge {
    int u, v;
    long long weight;
    
    bool operator<(const Edge& other) const {
        return weight < other.weight;
    }
};

// Disjoint Set Union (Union-Find)
int parent[MAXN], rankDSU[MAXN];

void initDSU(int n) {
    for (int i = 0; i < n; i++) {
        parent[i] = i;
        rankDSU[i] = 0;
    }
}

int find(int x) {
    if (parent[x] != x) {
        parent[x] = find(parent[x]);  // Path compression
    }
    return parent[x];
}

bool unite(int x, int y) {
    int px = find(x), py = find(y);
    if (px == py) return false;  // –í–µ—á–µ –≤ –µ–¥–Ω–æ –º–Ω–æ–∂–µ—Å—Ç–≤–æ
    
    // Union by rank
    if (rankDSU[px] < rankDSU[py]) swap(px, py);
    parent[py] = px;
    if (rankDSU[px] == rankDSU[py]) rankDSU[px]++;
    
    return true;
}

long long kruskal(vector<Edge>& edges, int n) {
    sort(edges.begin(), edges.end());  // –°–æ—Ä—Ç–∏—Ä–∞–Ω–µ –ø–æ —Ç–µ–≥–ª–æ
    initDSU(n);
    
    long long mstWeight = 0;
    vector<Edge> mstEdges;  // –†–µ–±—Ä–∞—Ç–∞ –≤ MST
    
    for (auto& e : edges) {
        if (unite(e.u, e.v)) {  // –ê–∫–æ –Ω–µ –æ–±—Ä–∞–∑—É–≤–∞—Ç —Ü–∏–∫—ä–ª
            mstWeight += e.weight;
            mstEdges.push_back(e);
            
            if (mstEdges.size() == n - 1) break;  // MST –µ –≥–æ—Ç–æ–≤–æ
        }
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ –≥—Ä–∞—Ñ—ä—Ç –µ —Å–≤—ä—Ä–∑–∞–Ω
    if (mstEdges.size() != n - 1) {
        return -1;  // –ù—è–º–∞ MST (–≥—Ä–∞—Ñ—ä—Ç –µ –Ω–µ—Å–≤—ä—Ä–∑–∞–Ω)
    }
    
    return mstWeight;
}
```

**–°–ª–æ–∂–Ω–æ—Å—Ç:** $O(E \log E)$ –∑–∞—Ä–∞–¥–∏ —Å–æ—Ä—Ç–∏—Ä–∞–Ω–µ—Ç–æ.

### 3.2. –ê–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –ü—Ä–∏–º (Prim)
–ü—Ä–∏–º –∏–∑–≥—Ä–∞–∂–¥–∞ MST –∏–Ω–∫—Ä–µ–º–µ–Ω—Ç–∞–ª–Ω–æ, –∑–∞–ø–æ—á–≤–∞–π–∫–∏ –æ—Ç –ø—Ä–æ–∏–∑–≤–æ–ª–µ–Ω –≤—Ä—ä—Ö –∏ –≤–∏–Ω–∞–≥–∏ –¥–æ–±–∞–≤—è–π–∫–∏ –Ω–∞–π-–µ–≤—Ç–∏–Ω–æ—Ç–æ —Ä–µ–±—Ä–æ, –∫–æ–µ—Ç–æ —Å–≤—ä—Ä–∑–≤–∞ –¥—ä—Ä–≤–æ—Ç–æ —Å –Ω–æ–≤ –≤—Ä—ä—Ö.

**–ü—ä–ª–Ω–∞ –∏–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è:**
```cpp
vector<pair<int, long long>> adj[MAXN];  // {—Å—ä—Å–µ–¥, —Ç–µ–≥–ª–æ}
bool inMST[MAXN];

long long prim(int start, int n) {
    fill(inMST, inMST + n, false);
    
    // Min-heap: {—Ç–µ–≥–ª–æ, {–≤—Ä—ä—Ö, –æ—Ç-–∫—ä–¥–µ}}
    priority_queue<pair<long long, pair<int, int>>,
                   vector<pair<long long, pair<int, int>>>,
                   greater<pair<long long, pair<int, int>>>> pq;
    
    long long mstWeight = 0;
    vector<pair<int, int>> mstEdges;
    
    // –ó–∞–ø–æ—á–≤–∞–º–µ –æ—Ç start
    pq.push({0, {start, -1}});
    
    while (!pq.empty()) {
        auto [w, vp] = pq.top();
        auto [v, from] = vp;
        pq.pop();
        
        if (inMST[v]) continue;  // –í–µ—á–µ –≤ MST
        
        inMST[v] = true;
        mstWeight += w;
        if (from != -1) {
            mstEdges.push_back({from, v});
        }
        
        // –î–æ–±–∞–≤—è–º–µ —Å—ä—Å–µ–¥–∏—Ç–µ –≤ –æ–ø–∞—à–∫–∞—Ç–∞
        for (auto [u, ew] : adj[v]) {
            if (!inMST[u]) {
                pq.push({ew, {u, v}});
            }
        }
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∫–∞ –¥–∞–ª–∏ –≤—Å–∏—á–∫–∏ –≤—ä—Ä—Ö–æ–≤–µ —Å–∞ –≤ MST
    for (int i = 0; i < n; i++) {
        if (!inMST[i]) return -1;  // –ì—Ä–∞—Ñ—ä—Ç –µ –Ω–µ—Å–≤—ä—Ä–∑–∞–Ω
    }
    
    return mstWeight;
}
```

**–°–ª–æ–∂–Ω–æ—Å—Ç:** $O(E \log V)$ —Å –ø—Ä–∏–æ—Ä–∏—Ç–µ—Ç–Ω–∞ –æ–ø–∞—à–∫–∞.

**–ö–æ–≥–∞ –¥–∞ –∏–∑–ø–æ–ª–∑–≤–∞–º–µ –∫–æ–π:**
- **–ö—Ä—É—Å–∫–∞–ª**: –ü–æ-–ª–µ—Å–µ–Ω –∑–∞ –∏–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è, –¥–æ–±—ä—Ä –∑–∞ —Ä–∞–∑—Ä–µ–¥–µ–Ω–∏ –≥—Ä–∞—Ñ–∏
- **–ü—Ä–∏–º**: –ü–æ-–µ—Ñ–µ–∫—Ç–∏–≤–µ–Ω –∑–∞ –ø–ª—ä—Ç–Ω–∏ –≥—Ä–∞—Ñ–∏, –ø–æ-–¥–æ–±—ä—Ä –∑–∞ –¥–∏–Ω–∞–º–∏—á–Ω–∏ –º–æ–¥–∏—Ñ–∏–∫–∞—Ü–∏–∏

## 4. –¢–æ–ø–æ–ª–æ–≥–∏—á–Ω–æ –°–æ—Ä—Ç–∏—Ä–∞–Ω–µ

–¢–æ–ø–æ–ª–æ–≥–∏—á–Ω–æ—Ç–æ —Å–æ—Ä—Ç–∏—Ä–∞–Ω–µ –ø–æ–¥—Ä–µ–∂–¥–∞ –≤—ä—Ä—Ö–æ–≤–µ—Ç–µ –Ω–∞ DAG (Directed Acyclic Graph) —Ç–∞–∫–∞, —á–µ –∑–∞ –≤—Å—è–∫–æ —Ä–µ–±—Ä–æ $u \to v$, $u$ —Å–µ –ø–æ—è–≤—è–≤–∞ –ø—Ä–µ–¥–∏ $v$ –≤ –ø–æ–¥—Ä–µ–¥–±–∞—Ç–∞. –°—ä—â–µ—Å—Ç–≤—É–≤–∞ —Å–∞–º–æ –∑–∞ –∞—Ü–∏–∫–ª–∏—á–Ω–∏ –≥—Ä–∞—Ñ–∏.

### 4.1. DFS –ø–æ–¥—Ö–æ–¥
–ò–∑–ø–æ–ª–∑–≤–∞–º–µ DFS –∏ –¥–æ–±–∞–≤—è–º–µ –≤—ä—Ä—Ö–æ–≤–µ—Ç–µ –≤ —Ä–µ–∑—É–ª—Ç–∞—Ç–∞ –ø—Ä–∏ –∏–∑–ª–∏–∑–∞–Ω–µ –æ—Ç —Ä–µ–∫—É—Ä—Å–∏—è—Ç–∞ (postorder), —Å–ª–µ–¥ —Ç–æ–≤–∞ –æ–±—Ä—ä—â–∞–º–µ —Ä–µ–¥–∞.

**–ò–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è:**
```cpp
vector<int> adj[MAXN];
bool visited[MAXN];
vector<int> topoOrder;

void topoDFS(int u) {
    visited[u] = true;
    
    for (int v : adj[u]) {
        if (!visited[v]) {
            topoDFS(v);
        }
    }
    
    topoOrder.push_back(u);  // –î–æ–±–∞–≤—è–º–µ –ø—Ä–∏ –∏–∑–ª–∏–∑–∞–Ω–µ
}

vector<int> topologicalSort(int n) {
    fill(visited, visited + n, false);
    topoOrder.clear();
    
    for (int i = 0; i < n; i++) {
        if (!visited[i]) {
            topoDFS(i);
        }
    }
    
    reverse(topoOrder.begin(), topoOrder.end());
    return topoOrder;
}
```

### 4.2. –ê–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –ö–∞–Ω (Kahn's Algorithm)
–ò–∑–ø–æ–ª–∑–≤–∞ –≤—Ö–æ–¥—è—â–∞—Ç–∞ —Å—Ç–µ–ø–µ–Ω (indegree) –Ω–∞ –≤—Å–µ–∫–∏ –≤—Ä—ä—Ö. –ó–∞–ø–æ—á–≤–∞–º–µ —Å –≤—ä—Ä—Ö–æ–≤–µ—Ç–µ —Å indegree 0 –∏ –ø–æ—Å—Ç–µ–ø–µ–Ω–Ω–æ "–ø—Ä–µ–º–∞—Ö–≤–∞–º–µ" —Ä–µ–±—Ä–∞—Ç–∞.

**–ò–º–ø–ª–µ–º–µ–Ω—Ç–∞—Ü–∏—è:**
```cpp
vector<int> adj[MAXN];
int indegree[MAXN];

vector<int> kahnTopologicalSort(int n) {
    fill(indegree, indegree + n, 0);
    
    // –ò–∑—á–∏—Å–ª—è–≤–∞–º–µ –≤—Ö–æ–¥—è—â–∞ —Å—Ç–µ–ø–µ–Ω
    for (int u = 0; u < n; u++) {
        for (int v : adj[u]) {
            indegree[v]++;
        }
    }
    
    queue<int> q;
    for (int i = 0; i < n; i++) {
        if (indegree[i] == 0) {
            q.push(i);
        }
    }
    
    vector<int> topoOrder;
    
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        topoOrder.push_back(u);
        
        // "–ü—Ä–µ–º–∞—Ö–≤–∞–º–µ" —Ä–µ–±—Ä–∞—Ç–∞ –æ—Ç u
        for (int v : adj[u]) {
            indegree[v]--;
            if (indegree[v] == 0) {
                q.push(v);
            }
        }
    }
    
    // –ê–∫–æ –Ω–µ —Å–º–µ –≤–∫–ª—é—á–∏–ª–∏ –≤—Å–∏—á–∫–∏ –≤—ä—Ä—Ö–æ–≤–µ, –∏–º–∞ —Ü–∏–∫—ä–ª
    if (topoOrder.size() != n) {
        return {};  // –ì—Ä–∞—Ñ—ä—Ç –∏–º–∞ —Ü–∏–∫—ä–ª
    }
    
    return topoOrder;
}
```

**–°–ª–æ–∂–Ω–æ—Å—Ç:** $O(V + E)$ –∑–∞ –¥–≤–∞—Ç–∞ –ø–æ–¥—Ö–æ–¥–∞.

**–ü—Ä–∏–ª–æ–∂–µ–Ω–∏—è:**
- –ü–ª–∞–Ω–∏—Ä–∞–Ω–µ –Ω–∞ –∑–∞–¥–∞—á–∏ —Å—ä—Å –∑–∞–≤–∏—Å–∏–º–æ—Å—Ç–∏

- –û–ø—Ä–µ–¥–µ–ª—è–Ω–µ –Ω–∞ —Ä–µ–¥–∞ –Ω–∞ –∫—É—Ä—Å–æ–≤–µ —Å prerequisites

**–ü—Ä–∏–º–µ—Ä:** –î–∞–¥–µ–Ω –µ —Å–ø–∏—Å—ä–∫ –æ—Ç –∫—É—Ä—Å–æ–≤–µ –∏ prerequesite-–∏. –í –∫–∞–∫—ä–≤ —Ä–µ–¥ —Ç—Ä—è–±–≤–∞ –¥–∞ –≤–∑–µ–º–µ—Ç–µ –∫—É—Ä—Å–æ–≤–µ—Ç–µ?

### 5.1. –ò–∑–±–æ—Ä –Ω–∞ –ø–æ–¥—Ö–æ–¥—è—â –∞–ª–≥–æ—Ä–∏—Ç—ä–º
**–¢–∞–±–ª–∏—Ü–∞ –∑–∞ –∏–∑–±–æ—Ä –Ω–∞ –∞–ª–≥–æ—Ä–∏—Ç—ä–º:**
