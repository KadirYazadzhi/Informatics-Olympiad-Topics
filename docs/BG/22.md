### –ó–∞–¥–∞—á–∞: Traveling Salesman Problem (TSP)

–ù–∞–º–∏—Ä–∞–Ω–µ –Ω–∞ –Ω–∞–π-–∫—Ä–∞—Ç–∫–∏—è –ø—ä—Ç, –∫–æ–π—Ç–æ –æ–±—Ö–æ–∂–¥–∞ –≤—Å–∏—á–∫–∏ –≥—Ä–∞–¥–æ–≤–µ —Ç–æ—á–Ω–æ –≤–µ–¥–Ω—ä–∂.

**–î–ü —Ä–µ—à–µ–Ω–∏–µ:** $dp[mask][i]$ = –º–∏–Ω–∏–º–∞–ª–Ω–∞—Ç–∞ —Ü–µ–Ω–∞ –¥–∞ –ø–æ—Å–µ—Ç–∏–º –≥—Ä–∞–¥–æ–≤–µ—Ç–µ –≤ –º–∞—Å–∫–∞—Ç–∞ –∏ –¥–∞ –∑–∞–≤—ä—Ä—à–∏–º –≤ –≥—Ä–∞–¥ $i$.

```cpp
const int INF = 1e9;
int n;  // –ë—Ä–æ–π –≥—Ä–∞–¥–æ–≤–µ
int dist[20][20];  // –†–∞–∑—Å—Ç–æ—è–Ω–∏—è –º–µ–∂–¥—É –≥—Ä–∞–¥–æ–≤–µ—Ç–µ
int dp[1 << 20][20];

int tsp() {
    for (int i = 0; i < (1 << n); i++)
        for (int j = 0; j < n; j++)
            dp[i][j] = INF;
    
    dp[1][0] = 0;  // –ó–∞–ø–æ—á–≤–∞–º–µ –æ—Ç –≥—Ä–∞–¥ 0
    
    for (int mask = 1; mask < (1 << n); mask++) {
        for (int i = 0; i < n; i++) {
            if (!(mask & (1 << i))) continue;  // i —Ç—Ä—è–±–≤–∞ –¥–∞ –µ –≤ –º–∞—Å–∫–∞—Ç–∞
            
            for (int j = 0; j < n; j++) {
                if (mask & (1 << j)) continue;  // j –Ω–µ —Ç—Ä—è–±–≤–∞ –¥–∞ –µ –ø–æ—Å–µ—Ç–µ–Ω
                
                int newMask = mask | (1 << j);
                dp[newMask][j] = min(dp[newMask][j], 
                                     dp[mask][i] + dist[i][j]);
            }
        }
    }
    
    // –í—Ä—ä—â–∞–º–µ —Å–µ –≤ –Ω–∞—á–∞–ª–Ω–∏—è –≥—Ä–∞–¥
    int ans = INF;
    int fullMask = (1 << n) - 1;
    for (int i = 0; i < n; i++) {
        ans = min(ans, dp[fullMask][i] + dist[i][0]);
    }
    return ans;
}
```
