```cpp
long long modInv(long long a, long long mod) {
    // –†–∞–∑—à–∏—Ä–µ–Ω –∞–ª–≥–æ—Ä–∏—Ç—ä–º –Ω–∞ –ï–≤–∫–ª–∏–¥

    while (a > 1) {
        long long q = a / mod;
        long long t = mod;
        mod = a % mod;
        a = t;
        t = x0;
        x0 = x1 - q * x0;
        x1 = t;
    }
    return (x1 % m0 + m0) % m0;
}

vector<long long> gaussianEliminationMod(vector<vector<long long>> A, 
                                          vector<long long> b) {
    int n = A.size();
    
    for (int col = 0; col < n; col++) {
        // –ù–∞–º–µ—Ä–∏ –Ω–µ–Ω—É–ª–µ–≤ –ø–∏–≤–æ—Ç
        int pivot = -1;
        for (int i = col; i < n; i++) {
            if (A[i][col] % MOD != 0) {
                pivot = i;
                break;
            }
        }
        
        if (pivot == -1) continue;
        
        swap(A[col], A[pivot]);
        swap(b[col], b[pivot]);
        
        long long inv = modInv(A[col][col], MOD);
        
        // –ï–ª–∏–º–∏–Ω–∏—Ä–∞–π
        for (int i = col + 1; i < n; i++) {
            long long factor = (A[i][col] * inv) % MOD;
            b[i] = ((b[i] - factor * b[col]) % MOD + MOD) % MOD;
            for (int j = col; j < n; j++) {
                A[i][j] = ((A[i][j] - factor * A[col][j]) % MOD + MOD) % MOD;
            }
        }
    }
    
    // –û–±—Ä–∞—Ç–Ω–∞ –∑–∞–º—è–Ω–∞ (–ø–æ–¥–æ–±–Ω–æ, –Ω–æ —Å –º–æ–¥—É–ª–Ω–∞ –∞—Ä–∏—Ç–º–µ—Ç–∏–∫–∞)
    vector<long long> x(n);
    for (int i = n - 1; i >= 0; i--) {
        x[i] = b[i];
        for (int j = i + 1; j < n; j++) {
            x[i] = ((x[i] - A[i][j] * x[j]) % MOD + MOD) % MOD;
        }
        x[i] = (x[i] * modInv(A[i][i], MOD)) % MOD;
    }
    
    return x;
}
```

### –†–∞–∑–±–∏—Ä–∞–Ω–µ –Ω–∞ XOR –∫–∞—Ç–æ –ª–∏–Ω–µ–π–Ω–∞ –∞–ª–≥–µ–±—Ä–∞

```cpp
struct XORBasis {
    static const int MAXLOG = 60;  // –ó–∞ 64-–±–∏—Ç–æ–≤–∏ —Ü–µ–ª–∏ —á–∏—Å–ª–∞
    long long basis[MAXLOG];
    int size;
    
    XORBasis() {
        memset(basis, 0, sizeof(basis));
        size = 0;
    }
    
    // –í–º—ä–∫–Ω–∏ —á–∏—Å–ª–æ –≤ –±–∞–∑–∏—Å–∞ (–≥–∞—É—Å–æ–≤–∞ –µ–ª–∏–º–∏–Ω–∞—Ü–∏—è)
    bool insert(long long x) {
        for (int i = MAXLOG - 1; i >= 0; i--) {
            if ((x & (1LL << i)) == 0) continue;
            
            if (basis[i] == 0) {
                basis[i] = x;
                size++;
                return true;  // x –µ –ª–∏–Ω–µ–π–Ω–æ –Ω–µ–∑–∞–≤–∏—Å–∏–º–æ
            }
            
            x ^= basis[i];  // –ï–ª–∏–º–∏–Ω–∏—Ä–∞–π –±–∏—Ç i
        }
        return false;  // x –µ –ª–∏–Ω–µ–π–Ω–æ –∑–∞–≤–∏—Å–∏–º–æ
    }
    
    // –ü—Ä–æ–≤–µ—Ä–∏ –¥–∞–ª–∏ —Å—Ç–æ–π–Ω–æ—Å—Ç—Ç–∞ –º–æ–∂–µ –¥–∞ –±—ä–¥–µ –ø—Ä–µ–¥—Å—Ç–∞–≤–µ–Ω–∞
    bool canRepresent(long long x) {
        for (int i = MAXLOG - 1; i >= 0; i--) {
            if ((x & (1LL << i)) == 0) continue;
            
            if (basis[i] == 0) return false;
            
            x ^= basis[i];
        }
        return x == 0;
    }
    
    // –í–∑–µ–º–∏ –º–∞–∫—Å–∏–º–∞–ª–Ω–∞ XOR —Å—Ç–æ–π–Ω–æ—Å—Ç
    long long getMax() {
        long long result = 0;
        for (int i = MAXLOG - 1; i >= 0; i--) {
            result = max(result, result ^ basis[i]);
        }
        return result;
    }
    
    // –í–∑–µ–º–∏ –º–∏–Ω–∏–º–∞–ª–Ω–∞ XOR —Å—Ç–æ–π–Ω–æ—Å—Ç
    long long getMin() {
        for (int i = 0; i < MAXLOG; i++) {
            if (basis[i] != 0) return basis[i];
        }
        return 0;
    }
    
    // –ë—Ä–æ–π —Ä–∞–∑–ª–∏—á–Ω–∏ XOR —Å—Ç–æ–π–Ω–æ—Å—Ç–∏ –≤—ä–∑–º–æ–∂–Ω–∏
    long long countDistinct() {
        return (1LL << size);
    }
};

    vector<long long> numbers = {3, 5, 7, 10};
    // –î–≤–æ–∏—á–Ω–æ: 0011, 0101, 0111, 1010
    
    for (long long x : numbers) {
        basis.insert(x);
    }

    cout << "–ú–∞–∫—Å XOR: " << basis.getMax() << "\n";
    cout << "–†–∞–∑–ª–∏—á–Ω–∏ —Å—Ç–æ–π–Ω–æ—Å—Ç–∏: " << basis.countDistinct() << "\n";
    
    return 0;
```

**–î–µ—Ç–µ—Ä–º–∏–Ω–∞–Ω—Ç–∞—Ç–∞** –µ —Å–∫–∞–ª–∞—Ä–Ω–∞ —Å—Ç–æ–π–Ω–æ—Å—Ç, –∫–æ—è—Ç–æ –∫–æ–¥–∏—Ä–∞ –≤–∞–∂–Ω–∏ —Å–≤–æ–π—Å—Ç–≤–∞ –Ω–∞ –∫–≤–∞–¥—Ä–∞—Ç–Ω–∞ –º–∞—Ç—Ä–∏—Ü–∞.

### –°–≤–æ–π—Å—Ç–≤–∞ –∏ –∑–Ω–∞—á–µ–Ω–∏–µ

- $\det(I) = 1$ (–µ–¥–∏–Ω–∏—á–Ω–∞ –º–∞—Ç—Ä–∏—Ü–∞)
- $\det(AB) = \det(A) \cdot \det(B)$
- $\det(A^T) = \det(A)$
- $\det(A) \neq 0 \iff$ –º–∞—Ç—Ä–∏—Ü–∞—Ç–∞ –µ –æ–±—Ä–∞—Ç–∏–º–∞
- $|\det(A)|$ –µ —Ñ–∞–∫—Ç–æ—Ä –Ω–∞ –º–∞—â–∞–±–∏—Ä–∞–Ω–µ –Ω–∞ –æ–±–µ–º–∞ –Ω–∞ –ª–∏–Ω–µ–π–Ω–æ—Ç–æ –ø—Ä–µ–æ–±—Ä–∞–∑–æ–≤–∞–Ω–∏–µ

### –ò–∑—á–∏—Å–ª—è–≤–∞–Ω–µ –Ω–∞ –¥–µ—Ç–µ—Ä–º–∏–Ω–∞–Ω—Ç–∞ —á—Ä–µ–∑ –≥–∞—É—Å–æ–≤–∞ –µ–ª–∏–º–∏–Ω–∞—Ü–∏—è

```cpp
double determinant(vector<vector<double>> A) {
    int n = A.size();
    double det = 1.0;
    
    for (int i = 0; i < n; i++) {
        // –ù–∞–º–µ—Ä–∏ –ø–∏–≤–æ—Ç
        int pivot = i;
        for (int j = i + 1; j < n; j++) {
            if (abs(A[j][i]) > abs(A[pivot][i])) {
                pivot = j;
            }
        }
        
        if (abs(A[pivot][i]) < EPS) {
            return 0;  // –°–∏–Ω–≥—É–ª—è—Ä–Ω–∞ –º–∞—Ç—Ä–∏—Ü–∞
        }
        
        if (pivot != i) {
            swap(A[i], A[pivot]);
            det *= -1;  // –†–∞–∑–º—è–Ω–∞—Ç–∞ –Ω–∞ —Ä–µ–¥–æ–≤–µ –ø—Ä–æ–º–µ–Ω—è –∑–Ω–∞–∫–∞
        }
        
        det *= A[i][i];
        
        // –ï–ª–∏–º–∏–Ω–∏—Ä–∞–π
        for (int j = i + 1; j < n; j++) {
            double factor = A[j][i] / A[i][i];
            for (int k = i; k < n; k++) {
                A[j][k] -= factor * A[i][k];
            }
        }
    }
    
    return det;
}
```

```cpp
long long determinantMod(vector<vector<long long>> A, long long mod) {
    int n = A.size();
    long long det = 1;
    
    for (int i = 0; i < n; i++) {
        int pivot = -1;
        for (int j = i; j < n; j++) {
            if (A[j][i] % mod != 0) {
                pivot = j;
                break;
            }
        }
        
        if (pivot == -1) return 0;
        
        if (pivot != i) {
            swap(A[i], A[pivot]);
            det = (mod - det) % mod;
        }
        
        det = (det * A[i][i]) % mod;
        long long inv = modInv(A[i][i], mod);
        
        for (int j = i + 1; j < n; j++) {
            long long factor = (A[j][i] * inv) % mod;
            for (int k = i; k < n; k++) {
                A[j][k] = ((A[j][k] - factor * A[i][k]) % mod + mod) % mod;
            }
        }
    }
    
    return det;
}
```

```cpp
int gaussianEliminationBinary(int n, int m) {

    for (int col = 0; col < m && rank < n; col++) {
        int pivot = -1;
        for (int i = rank; i < n; i++) {
            if (matrix[i][col]) {
                pivot = i;
                break;
            }
        }
        
        if (pivot == -1) continue;
```
