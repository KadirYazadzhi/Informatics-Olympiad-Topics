bool isSafe(const std::vector<std::vector<int>>& board, int row, int col) {
    for (int i = 0; i < col; i++)

    for (int i = row, j = col; i >= 0 && j >= 0; i--, j--)

bool solveNQueens(std::vector<std::vector<int>>& board, int col) {
